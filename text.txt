#include <WiFi.h>
#include <PubSubClient.h>

#include <Arduino.h>
#include <U8g2lib.h>
#include <Wire.h>

U8X8_SH1106_128X64_NONAME_HW_I2C u8x8(/* reset=*/ -1, /* clock=*/ 9, /* data=*/ 8);

const char* ssid = "OpenSpace322";
const char* password = "LAN322LAN";
//const char* mqtt_server = "broker.hivemq.com"; 
//const char* mqtt_server = "3.121.214.225";
const char* mqtt_server = "broker.emqx.io";
const char* topic = "ukraine/esp32/Wi-Fi_module";

const int8_t motors[4] = {38, 37, 40, 39};

WiFiClient espClient;
PubSubClient client(espClient);

void callback(char* topic, byte* payload, unsigned int length) {
  Serial.print("Зловлено сигнал з топіку: ");
  Serial.println(topic);

  // -- motors initialization --
  pinMode(motors[0], OUTPUT);
  pinMode(motors[1], OUTPUT);
  pinMode(motors[2], OUTPUT);
  pinMode(motors[3], OUTPUT);

  String message = "";
  for (int i = 0; i < length; i++) { 
    message += (char)payload[i]; 
  }
  
  Serial.print("Текст повідомлення: ");
  Serial.println(message);
  
  if (message == "leftward") {
    Serial.println("УРА! Команда 'leftward' успішно оброблена!"); 
    Motors(0, 20, 80, 0);
  }

  if (message == "forward") {
    Serial.println("УРА! Команда 'forward' успішно оброблена!"); 
    Motors(80, 0, 80, 0);
  }

  if (message == "backward") {
    Serial.println("УРА! Команда 'backward' успішно оброблена!"); 
    Motors(0, 80, 0, 80);
  }

  if (message == "rightward_go") {
    Serial.println("УРА! Команда 'rightward_go' успішно оброблена!"); 
    //Motors(80, 0, 0, 20);
  }else if (message == "rightward_stop"){

    Serial.println("УРА! Команда 'rightward_stop' успішно оброблена!"); 
    //Motors(0, 0, 0, 0);

  }else{

    Serial.println("Щось пішло не так rightward!");

  }
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Спроба підключення до MQTT брокера... ");
    
    
    String clientId = "ESP32_Client_" + String(random(0, 10000)); 
    
    if (client.connect(clientId.c_str())) { 
      Serial.println("[УСПІШНО]");
      client.subscribe(topic); 
      Serial.print("Підписалися на топік: ");
      Serial.println(topic);
      u8x8.drawString(1, 4, "Ready to work!");
    } else {
      Serial.print("[ПОМИЛКА], код помилки = ");
      Serial.print(client.state());
      Serial.println(", повтор через 5 секунд...");
      u8x8.drawString(0, 4, "Waiting connection...");
      delay(5000);
      u8x8.clear();
    }
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000); 
  Serial.println("\n--- Старт програми ---");

  u8x8.begin();
  u8x8.setFlipMode(1);
  u8x8.setPowerSave(0); 

  u8x8.setFont(u8x8_font_chroma48medium8_r);
  u8x8.clear();
  
  Serial.print("Підключення до Wi-Fi: ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);
  
  while (WiFi.status() != WL_CONNECTED) { 
    delay(500); 
    Serial.print("."); 
  }
  
  Serial.println("\n[Wi-Fi ПІДКЛЮЧЕНО!]");
  Serial.print("IP адреса плати: ");
  Serial.println(WiFi.localIP());
  
  client.setServer(mqtt_server, 1883);
  client.setCallback(callback);
}

void loop() {
  if (!client.connected()) { 
    reconnect(); 
  }
  client.loop(); 
}

void Motors(int moto1_forward, int moto1_backward, int moto2_forward, int moto2_backward){

  analogWrite(motors[0], moto1_forward);
  analogWrite(motors[1], moto1_backward);
  analogWrite(motors[2], moto2_forward);
  analogWrite(motors[3], moto2_backward);

  delay(500);

  analogWrite(motors[0], 0);
  analogWrite(motors[1], 0);
  analogWrite(motors[2], 0);
  analogWrite(motors[3], 0);

}
